Singleton getinstance method block must be synchronized

Noncompliant Code Example


	    public static Singleton getInstance(int pp) {
        if (instance == null) {
                                              // Noncompliant {{Synchronize the getinstance method}}
                if (instance == null) {
                    instance = new Singleton();
                }
            
        }

        return instance;
    }

Compliant Solution

	     public static Singleton getInstance(int pp) {
        if (instance == null) {
            synchronized(Singleton.class) {
                if (instance == null) {
                    instance = new Singleton();
                }
            }
        }

        return instance;
    }
